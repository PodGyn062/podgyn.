<script>
const WHATS = "5562981808194";

const products = [

["Ignite V80",100],

["Ignite V120 Pro",120],
["Ignite V155 Ultra Slim",130],

["Ignite V300 Ultra Slim",170],

["Ignite Sweet",180],
["Ignite Mix",180],
["Elfbar 5K",90],
["Elfbar 15K",120],
["Elfbar BC15K",120],
["Elfbar GH23K",140],
["Elfbar TE30K",160],
["Elfbar Trio",170],
["Elfbar Ice King",180],
["Geek Bar Pulse",170],
["Geek Bar 25K",170],
["Lost Mary MO10K",110],
["Lost Mary MO20K Pro",170],
["Lost Mary Mixer+",180],
["Lost Angel",140],
["Oxbar 50K",220],
["Waka 15K",120],
["Waka 50K",220],
["Nick Bar 10K",110],
["Pro 30K",160]
];

let cart = [];

function renderProducts(){
  let el = document.getElementById("products");
  el.innerHTML = "";

  products.forEach(p=>{
    el.innerHTML += `
      <div class="card" data-name="${p[0]}">
        <h4>${p[0]}</h4>
        <p>R$ ${p[1]}</p>

        <small>Sabores e disponibilidade via WhatsApp</small>

        <button class="btn-cart" onclick="add('${p[0]}',${p[1]})">
          Adicionar
        </button>

        <button class="btn-whats" onclick="askWhats('${p[0]}')">
          Ver no WhatsApp
        </button>
      </div>
    `;
  });
}
renderProducts();

function add(name,price){
  let item = cart.find(i=>i.name===name);
  if(item) item.qtd++;
  else cart.push({name,price,qtd:1});
  render();
}

function plus(i){ cart[i].qtd++; render(); }
function minus(i){ cart[i].qtd--; if(cart[i].qtd<=0) cart.splice(i,1); render(); }

function render(){
  let el = document.getElementById("cart");
  let total = 0;
  el.innerHTML = "";

  cart.forEach((i,index)=>{
    total += i.price * i.qtd;

    el.innerHTML += `
      <div>
        ${i.name} x${i.qtd}
        <button onclick="minus(${index})">-</button>
        <button onclick="plus(${index})">+</button>
      </div>
    `;
  });

  let taxa = parseFloat(document.getElementById("taxa").value || 0);
  total += taxa;

  document.getElementById("total").innerText = "Total: R$ " + total.toFixed(2);
}

function filter(){
  let v = document.getElementById("search").value.toLowerCase();
  document.querySelectorAll(".card").forEach(c=>{
    c.style.display = c.dataset.name.toLowerCase().includes(v) ? "block" : "none";
  });
}

function askWhats(produto){
  let msg = "Olá, quero saber sabores e disponibilidade de: " + produto;
  let url = "https://wa.me/" + WHATS + "?text=" + encodeURIComponent(msg);
  window.open(url, "_blank");
}

function send(){
  let nome = document.getElementById("nome").value;
  let taxa = parseFloat(document.getElementById("taxa").value || 0);

  let msg = `Pedido - ${nome}%0A`;
  let total = 0;

  cart.forEach(i=>{
    msg += `- ${i.name} x${i.qtd} = R$ ${i.price*i.qtd}%0A`;
    total += i.price*i.qtd;
  });

  total += taxa;

  msg += `%0ATaxa: R$ ${taxa}`;
  msg += `%0ATotal: R$ ${total.toFixed(2)}`;

  let url = "https://wa.me/" + WHATS + "?text=" + encodeURIComponent(msg);
  window.open(url, "_blank");
}
